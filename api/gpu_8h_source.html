44struct Shape {

46 // dynamic allocation for shape data

48 size_t rank = 0;

734inline void toCPU(Context &ctx, Tensor &tensor, void *data, size_t bufferSize,
735 CopyData &op) {
736 wgpuQueueSubmit(ctx.queue, 1, &op.commandBuffer);
737 CallbackData callbackData = {op.readbackBuffer, bufferSize, data, &op.promise,
738 &op.future};
739 wgpuQueueOnSubmittedWorkDone(
740 ctx.queue,
741 [](WGPUQueueWorkDoneStatus status, void *callbackData) {
742 check(status == WGPUQueueWorkDoneStatus_Success, "Queue work done",
743 __FILE__, __LINE__);
744 const auto *data = static_cast<CallbackData *>(callbackData);
745 wgpuBufferMapAsync(
746 data->buffer, WGPUMapMode_Read, 0, data->bufferSize,
747 [](WGPUBufferMapAsyncStatus status, void *captureData) {
748 const auto *data = static_cast<CallbackData *>(captureData);
749 check(status == WGPUBufferMapAsyncStatus_Success,
750 "Map readbackBuffer", __FILE__, __LINE__);
751 const void *mappedData = wgpuBufferGetConstMappedRange(
752 data->buffer, /*offset=*/0, data->bufferSize);
753 check(mappedData, "Get mapped range", __FILE__, __LINE__);
754 memcpy(data->output, mappedData, data->bufferSize);
755 wgpuBufferUnmap(data->buffer);
756 data->promise->set_value();
757 },
758 callbackData);
759 },
760 &callbackData);
761 wait(ctx, op.future);
762}

778inline void toCPU(Context &ctx, Tensor &tensor, void *data, size_t bufferSize) {
779 CopyData op;
780 op.future = op.promise.get_future();
781 {
782 WGPUBufferDescriptor readbackBufferDescriptor = {
783 .usage = WGPUBufferUsage_CopyDst | WGPUBufferUsage_MapRead,
784 .size = bufferSize,
785 };
786 op.readbackBuffer =
787 wgpuDeviceCreateBuffer(ctx.device, &readbackBufferDescriptor);
788 }
789 {
790 WGPUCommandEncoder commandEncoder;
791 WGPUComputePassEncoder computePassEncoder;
792 commandEncoder = wgpuDeviceCreateCommandEncoder(ctx.device, nullptr);
793 wgpuCommandEncoderCopyBufferToBuffer(commandEncoder, tensor.data.buffer, 0,
794 op.readbackBuffer, 0, bufferSize);
795 op.commandBuffer = wgpuCommandEncoderFinish(commandEncoder, nullptr);
796 check(op.commandBuffer, "Create command buffer", __FILE__, __LINE__);
797 }
798 toCPU(ctx, tensor, data, bufferSize, op);

880inline void resetCommandBuffer(WGPUDevice &device, Kernel &op) {
881 {
882 WGPUCommandEncoder commandEncoder =
883 wgpuDeviceCreateCommandEncoder(device, nullptr);
884 WGPUComputePassEncoder computePassEncoder =

950inline Kernel createKernel(Context &ctx, const KernelCode &code,
951 const Tensor *dataBindings, size_t numTensors,
952 const size_t *viewOffsets, const Shape &nWorkgroups,
953 const void *params = nullptr,
954 size_t paramsSize = 0) {
955 assert(nWorkgroups.rank == 3);
956 WGPUDevice device = ctx.device;
957 WGPUQueue queue = ctx.queue;
958 Kernel op;
959 // paramIndex is the index into bgLayoutEntries for the parameters buffer If
960 // there are no parameters for the kernel, paramsSize == 0 and paramIndex is
961 // effectively undefined (== -1)
962 size_t paramIndex = -1;
963 // Note: paramIndex is undefined unless paramsSize > 0
964 size_t numBindings = numTensors;
965 if (paramsSize > 0) {
966 numBindings++; // parameters buffer
967 paramIndex = numBindings - 1; // index of the parameters buffer within
968 // op.buffers, op.bufferSizes and
969 // bgLayoutEntries
970 }
971 op.buffers = std::make_unique<WGPUBuffer[]>(numBindings);
972 op.bufferSizes = std::make_unique<size_t[]>(numBindings);
973 op.numBindings = numBindings;
974 std::vector<WGPUBindGroupLayoutEntry> bgLayoutEntries(numBindings);
975 // Create layout entries for input buffers
976 for (size_t i = 0; i < numTensors; ++i) {
977 bgLayoutEntries[i] = WGPUBindGroupLayoutEntry{
978 .binding = static_cast<uint32_t>(i),
979 .visibility = WGPUShaderStage_Compute,
980 .buffer =
981 WGPUBufferBindingLayout{
982 .type = WGPUBufferBindingType_Storage,
983 .minBindingSize = dataBindings[i].data.size,
984 },
985 };
986 }
987 if (paramsSize > 0) {
988 LOG(kDefLog, kInfo, "Create layout entry for the params buffer");
989 // Create layout entry for the params buffer
990 bgLayoutEntries[paramIndex] = WGPUBindGroupLayoutEntry{
991 .binding = static_cast<uint32_t>(paramIndex),
992 .visibility = WGPUShaderStage_Compute,
993 .buffer =
994 WGPUBufferBindingLayout{
995 .type = WGPUBufferBindingType_Uniform,
996 .minBindingSize = paramsSize,
997 },
998 };
999 }
1000 WGPUBindGroupLayoutDescriptor bgLayoutDesc = {
1001 .entryCount = static_cast<uint32_t>(bgLayoutEntries.size()),
1002 .entries = bgLayoutEntries.data(),
1003 };
1004 WGPUBindGroupLayout bgLayout =
1005 wgpuDeviceCreateBindGroupLayout(device, &bgLayoutDesc);
1006 for (size_t i = 0; i < numTensors; ++i) {
1007 op.buffers[i] = dataBindings[i].data.buffer;
1008 op.bufferSizes[i] = dataBindings[i].data.size;
1009 }
1010 // Create a buffer for the Params struct
1011 if (paramsSize > 0) {
1012 WGPUBufferDescriptor paramsBufferDesc = {
1013 .usage = WGPUBufferUsage_Uniform | WGPUBufferUsage_CopyDst,
1014 .size = paramsSize,
1015 .mappedAtCreation = false,
1016 };
1017 op.buffers[paramIndex] = wgpuDeviceCreateBuffer(device, &paramsBufferDesc);
1018 op.bufferSizes[paramIndex] = paramsSize;
1019 wgpuQueueWriteBuffer(queue, op.buffers[paramIndex], 0, params, paramsSize);
1020 LOG(kDefLog, kTrace, "Params buffer written");
1021 } else {
1022 LOG(kDefLog, kTrace, "No params buffer needed");
1023 }
1024 std::vector<WGPUBindGroupEntry> bindGroupEntries(numBindings);
1025 for (size_t i = 0; i < numTensors; ++i) {
1026 bindGroupEntries[i] = WGPUBindGroupEntry{
1027 .binding = static_cast<uint32_t>(i),
1028 .buffer = op.buffers[i],
1029 .offset = viewOffsets[i],
1030 .size = op.bufferSizes[i],
1031 };
1032 }
1033 if (paramsSize > 0) {
1034 LOG(kDefLog, kInfo, "Create bind group entry for the params buffer");
1035 LOG(kDefLog, kInfo, "paramIndex: %d", paramIndex);
1036 bindGroupEntries[paramIndex] = WGPUBindGroupEntry{
1037 .binding = static_cast<uint32_t>(paramIndex),
1038 .buffer = op.buffers[paramIndex],
1039 .offset = 0,
1040 .size = paramsSize,
1041 };
1042 }
1043 LOG(kDefLog, kTrace, "BG Entries Size: %d", numBindings);
1044 WGPUBindGroupDescriptor bindGroupDesc = {
1045 .layout = bgLayout,
1046 .entryCount = static_cast<uint32_t>(numBindings),
1047 .entries = bindGroupEntries.data(),
1048 };
1049 op.bindGroup = wgpuDeviceCreateBindGroup(device, &bindGroupDesc);
1050 {
1051 WGPUPipelineLayoutDescriptor pipelineLayoutDesc = {
1052 .bindGroupLayoutCount = 1,
1053 .bindGroupLayouts = &bgLayout,
1054 };
1055 WGPUPipelineLayout pipelineLayout =
1056 wgpuDeviceCreatePipelineLayout(device, &pipelineLayoutDesc);
1057 WGPUShaderModuleWGSLDescriptor wgslDesc = {
1058 .code = code.data.c_str(),
1059 };
1060 wgslDesc.chain.sType = WGPUSType_ShaderModuleWGSLDescriptor;
1061 WGPUShaderModuleDescriptor shaderModuleDesc = {};
1062 shaderModuleDesc.nextInChain = &wgslDesc.chain;
1063 shaderModuleDesc.label = code.label.c_str();
1064 WGPUComputePipelineDescriptor computePipelineDesc = {};
1065 computePipelineDesc.layout = pipelineLayout;
1066 computePipelineDesc.compute.module =
1067 wgpuDeviceCreateShaderModule(device, &shaderModuleDesc);
1068 computePipelineDesc.compute.entryPoint = code.entryPoint.c_str();
1069 computePipelineDesc.label = code.label.c_str();
1070 op.computePipeline =
1071 wgpuDeviceCreateComputePipeline(device, &computePipelineDesc);
1072 }
1073 /*
1074 op.nWorkgroups = {cdiv(nThreads[0], code.workgroupSize[0]),
1075 cdiv(nThreads[1], code.workgroupSize[1]),
1076 cdiv(nThreads[2], code.workgroupSize[2])};
1077 */
1078 op.nWorkgroups = {nWorkgroups[0], nWorkgroups[1], nWorkgroups[2]};
